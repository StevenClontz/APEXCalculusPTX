\begin{align*} \int_{C_2} \vec F\cdot d\vec r_2 \amp = \int_{\pi}^{2\pi} \la \sin t+1,\cos t\ra\cdot\la -\sin t,\cos t\ra\, dt\\ \amp = \int_{\pi}^{2\pi} \big(-\sin^2t-\sin t+\cos^2t\big)\, dt = 2 \text{ ft-lbs } \text{.} \end{align*}
